module booth_multiplier_16bit ( 
    input  signed [15:0] multiplicand, 
    input  signed [15:0] multiplier, 
    output reg signed [31:0] product 
); 
 
    integer i; 

    reg signed [32:0] A; 
    reg signed [16:0] M; 
    reg signed [16:0] Q; 
 
    always @(*) begin 
        A = 33'sd0; 
        M = {multiplicand, 1'b0}; 
        Q = {multiplier, 1'b0}; 
 
        for (i = 0; i < 16; i = i + 1) begin 
            case (Q[1:0]) 
                2'b01: A = A + {{16{M[16]}}, M}; 
                2'b10: A = A - {{16{M[16]}}, M}; 
                default: A = A; 
            endcase 
 
            {A, Q} = $signed({A, Q}) >>> 1; 
        end 
 
        product = {A[15:0], Q[16:1]}; 
    end 
 
endmodule
